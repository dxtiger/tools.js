CorePlugin

TE

渲染方法

TE

创建一个实例

调用:

TE();

返回:

  • object-(object) 模板渲染实例

例子:

javascript
var a = TE();
console.log(a); // object

TE.setTag

设置模板的界定符

调用:

TE.setTag(open,close);

参数:

  1. open-(string) 开始符
  2. close-(string) 结束符

例子:

javascript
var a = TE();

a.setTag('{@','@}');

TE.assign

写入数据

调用:

TE.assign(key,value);
TE.assign(options);

参数:

  1. key-(string) 变量名
  2. value 值
  3. options-(object) json格式的样式及样式值. {key:value}
    • key-(string) 变量名
    • value 值

例子:

javascript
var a = TE();

a.assign('name','tiger');
a.assign({
	'aaa' : '111',
	'bbb' : '222'
})

TE.display

渲染模板

调用:

TE.display(templete);

参数:

  1. templete-(string) 模板

返回:

  1. html-(string) 渲染完成的html字符串

例子:

javascript
var a = TE();

var tem = '<div><@= name @></div>';

a.assign('name','tiger');

var html = a.display(tem);
console.log(html); // <div>tiger</div>


					
var b = TE();

var tem2 = '<ul>'+
           '<@ for(var key in data){ @>' +
              '<li><@= key @> : <@= data[key]></li>' + 
           '<@ } @>'
           '</ul>'

b.assign({
	'aaa' : '111',
	'bbb' : '222',
	'ccc' : '333'
});

var str = b.display(tem2);
console.log(str); // <ul><li>aaa : 111</li><li>bbb : 222</li><li>ccc : 333</li></ul>

TE.render

渲染模板

调用:

TE.render(templete,data);

参数:

  1. templete-(string) 模板
  2. data-(object) 数据,json格式
    • key-(string) 变量名
    • value 值

返回:

  1. html-(string) 渲染完成的html字符串

例子:

javascript
var a = TE();

var tem = '<div><@= name @></div>';

var html = a.render(tem,{'name':'tiger'});
console.log(html); // <div>tiger</div>


					
var b = TE();

var tem2 = '<ul>'+
           '<@ for(var key in data){ @>' +
               '<li><@= key @> : <@= data[key]></li>' + 
           '<@ } @>'
           '</ul>'

var str = b.render(tem2,{ data : {
	'aaa' : '111',
	'bbb' : '222',
	'ccc' : '333'
    }
});

console.log(str); // <ul><li>aaa : 111</li><li>bbb : 222</li><li>ccc : 333</li></ul>
