CorePlugin

Event

时间监听方法:添加事件,删除事件,阻止事件,获取事件源

Event.stop

阻止事件

调用:

Event.stop(e);

参数:

  1. e-(object) 事件

例子:

html
<a id="link" href="http://www.google.com" class="name"></a>
javascript
var link = document.getElementById('link');
link.onclick = function(e){
	Event.stop(e);
	
}

Event.target

获取事件源节点

调用:

Event.target(e);

参数:

  1. e-(object) 事件

返回:

  • element (object) 事件源节点

例子:

html
<a id="link" href="http://www.google.com" class="name"></a>
javascript
var link = document.getElementById('link');
document.body.onclick = function(e){
	Event.stop(e);
	var tar = Event.target(e);
	console.log(tar)
}

// 点击 link, 显示信息  <a id="link" href="http://www.google.com" class="name"></a>

Event.add

添加事件

调用:

Event.add(obj,type,fn);

参数:

  1. obj-(object) dom节点
  2. type-(string) 事件类型.click,mouseover,mousout,keydown等
  3. fn-(function) 方法

例子:

html
<a id="link" href="http://www.google.com" class="name"></a>
javascript
var link = document.getElementById('link');
function linkFn(e){
	Event.stop(e);
	console.log(Event.target(e));
}
Event.add(link,'click',linkFn);

// 点击 link, 显示信息  <a id="link" href="http://www.google.com" class="name"></a>

Event.remove

删除事件

调用:

Event.remove(obj,type,fn);

参数:

  1. obj-(object) dom节点
  2. type-(string) 事件类型.click,mouseover,mousout,keydown等
  3. fn-(function) 方法

例子:

html
<a id="link" href="http://www.google.com" class="name"></a>
javascript
var link = document.getElementById('link');
function linkFn(e){
	Event.stop(e);
	console.log(Event.target(e));
}

Event.add(link,'click',linkFn);
// 点击 link, 显示信息  <a id="link" href="http://www.google.com" class="name"></a>

Event.remove(link,'click',linkFn);
// 点击 link, 无信息显示. 事件已删除.


Event.add(link,'mouseover',function(e){
	linkFn(e);
})
// 滑过 link, 显示信息  <a id="link" href="http://www.google.com" class="name"></a>


Event.remove(link,'mouseover',function(e){
	linkFn(e);
})
// 滑过 link, 显示信息  <a id="link" href="http://www.google.com" class="name"></a>

// 添加的是一个匿名方法,无法找到此方法,所以无法删除
